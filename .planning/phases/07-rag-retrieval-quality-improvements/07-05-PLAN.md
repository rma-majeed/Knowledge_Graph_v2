```python
"""Feature flags for RAG retrieval quality improvements (Phase 7).

All flags are read from environment variables at module import time.
Use importlib.reload(retrieval_config) in tests after monkeypatching env vars.

Defaults:
  RAG_ENABLE_BM25       = True   (on by default — addresses vocabulary mismatch)
  RAG_ENABLE_RERANKER   = True   (on by default — requires sentence-transformers)
  RAG_ENABLE_PARENT_DOC = False  (opt-in — requires re-ingest to populate chunk_parents)
  RAG_ENABLE_ENRICHMENT = False  (opt-in — requires re-ingest with LM Studio running)

Override in .env or shell:
    RAG_ENABLE_BM25=false         # disable BM25 (fallback to pure vector)
    RAG_ENABLE_RERANKER=false     # disable cross-encoder (faster but lower quality)
    RAG_ENABLE_PARENT_DOC=true    # enable after re-ingesting documents
    RAG_ENABLE_ENRICHMENT=true    # enable before first ingest of a corpus
"""
from __future__ import annotations

import os


def _flag(name: str, default: bool) -> bool:
    """Read a boolean feature flag from environment variable.

    Accepts: "true"/"1"/"yes" for True; anything else (including unset) for default.
    Case-insensitive.
    """
    val = os.getenv(name)
    if val is None:
        return default
    return val.strip().lower() in ("true", "1", "yes")


RAG_ENABLE_BM25: bool = _flag("RAG_ENABLE_BM25", True)
RAG_ENABLE_RERANKER: bool = _flag("RAG_ENABLE_RERANKER", True)
RAG_ENABLE_PARENT_DOC: bool = _flag("RAG_ENABLE_PARENT_DOC", False)
RAG_ENABLE_ENRICHMENT: bool = _flag("RAG_ENABLE_ENRICHMENT", False)
```
